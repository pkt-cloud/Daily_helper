$ErrorActionPreference = "Stop"

# Version and paths
$GitTfsVersion = "0.34.0"
$InstallRoot = "$env:LOCALAPPDATA\Programs\git-tfs"
$ZipPath = "$env:TEMP\GitTfs-$GitTfsVersion.zip"
$DownloadUrl = "https://github.com/git-tfs/git-tfs/releases/download/v$GitTfsVersion/GitTfs-$GitTfsVersion.zip"

# Force TLS 1.2 for older Windows/PowerShell environments
[Net.ServicePointManager]::SecurityProtocol = [Net.SecurityProtocolType]::Tls12

# Clean previous install
if (Test-Path $InstallRoot) {
    Remove-Item $InstallRoot -Recurse -Force
}

New-Item -ItemType Directory -Force -Path $InstallRoot | Out-Null

# Download using curl.exe because it is more reliable than Invoke-WebRequest in many corporate networks
curl.exe -L $DownloadUrl -o $ZipPath

# Validate download
if (!(Test-Path $ZipPath)) {
    throw "Download failed. ZIP file was not created: $ZipPath"
}

# Extract
Expand-Archive -Path $ZipPath -DestinationPath $InstallRoot -Force

# Find git-tfs.exe even if ZIP extracted into nested folder
$GitTfsExe = Get-ChildItem -Path $InstallRoot -Filter "git-tfs.exe" -Recurse | Select-Object -First 1

if (-not $GitTfsExe) {
    throw "git-tfs.exe not found after extraction. Check ZIP extraction under: $InstallRoot"
}

$GitTfsFolder = $GitTfsExe.Directory.FullName

# Add to current PowerShell session PATH
$env:Path = "$GitTfsFolder;$env:Path"

# Add to user PATH permanently
$UserPath = [Environment]::GetEnvironmentVariable("Path", "User")
if ($UserPath -notlike "*$GitTfsFolder*") {
    [Environment]::SetEnvironmentVariable("Path", "$UserPath;$GitTfsFolder", "User")
}

Write-Host "git-tfs installed at: $GitTfsFolder"
Write-Host "Testing direct exe..."
& "$GitTfsFolder\git-tfs.exe" --version

Write-Host "Testing Git alias..."
git tfs --version
